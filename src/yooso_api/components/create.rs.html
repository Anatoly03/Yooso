35/// TODO: document
36#[post("/", data = "<body>")]
37pub async fn create_component(
38    state: &State<MetaDBState>,
39    general_state: &State<GeneralDBState>,
40    body: Json<CreateComponentRequest>,
41) -> Result<Json<CreateComponentResponse>> {
42    let uuid = uuid::Uuid::now_v7();
43    let created_at = chrono::Utc::now().timestamp_millis();
44
45    // Create new Component
46    let new_component = ComponentTable {
47        id: uuid,
48        component_name: body.name.clone(),
49        is_system: body.is_system,
50        color: body.color,
51        created_at,
52    };
53
54    // Validate component metadata before saving to the database.
55    new_component.validate()?;
56
57    let (new_fields, errors) = body
58        .fields
59        .iter()
60        .enumerate()
61        .map(|(position, field)| {
62            let field = ComponentFieldTable {
63                id: uuid::Uuid::now_v7(),
64                component_id: uuid,
65                field_name: field.name.clone(),
66                field_type: field.field_type.clone(),
67                is_system: field.is_system,
68                position: position as i32,
69                created_at,
70            };
71
72            // Validate field metadata before saving to the database.
73            field.validate()?;
74
75            Ok(field)
76        })
77        .fold((vec![], vec![]), |(mut fields, mut errors), field_result| {
78            match field_result {
79                Ok(field) => fields.push(field),
80                Err(e) => errors.push(e),
81            }
82            (fields, errors)
83        });
84
85    // If there are any validation errors, return the first one.
86    if let Some(error) = errors.into_iter().next() {
87        return Err(error);
88    }
89
90    // Save component and fields to the metadata database.
91    new_component.save(state).await?;
92
93    for field in &new_fields {
94        field.save(state).await?;
95    }
96
97    // Save component schema to the general database.
98    new_component.create_dynamic_table(general_state, &new_fields).await?;
99
100    // Return the created component and fields in the response.
101    let fields = new_fields
102        .iter()
103        .map(|field| ComponentField {
104            id: field.id,
105            name: field.field_name.clone(),
106            field_type: field.field_type.clone(),
107            is_system: field.is_system,
108            created_at: field.created_at,
109        })
110        .collect::<Vec<_>>();
111
112    Ok(Json(CreateComponentResponse {
113        success: true,
114        metadata: Component {
115            id: new_component.id,
116            name: new_component.component_name,
117            is_system: new_component.is_system,
118            color: new_component.color,
119            created_at: new_component.created_at,
120        },
121        fields,
122    }))
123}
