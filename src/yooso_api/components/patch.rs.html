23#[derive(Debug, Clone, Serialize, Deserialize)]
24#[serde(tag = "operation")]
25pub enum PatchFieldRequest {
26    #[serde(rename = "add")]
27    AddField {
28        name: String,
29        is_system: bool,
30        field_type: String,
31    },
32    #[serde(rename = "update")]
33    UpdateField {
34        id: Uuid,
35        name: String,
36        is_system: bool,
37        field_type: String,
38    },
39    #[serde(rename = "remove")]
40    DeleteField { id: Uuid },
41}
