1//! This module contains methods for parsing an [ItemStruct] and converting
2//! fields metadata into a usable format for the [crate::collection] macro.
3//!
4//! # Rust-SQL Type Mapping
5//!
6//! See the [Type Affinity in SQLite](https://sqlite.org/datatype3.html#type_affinity)
7//!
8//! | Rust Type | SQL Type |
9//! |-----------|----------|
10//! | `i32`, `i64` | `INTEGER` |
11//! | `f32`, `f64` | `REAL` |
12//! | `String` | `TEXT` |
13//! | `bool` | `BOOLEAN` |
14//! | `Uuid` | `TEXT` |
15//! | `Vec<u8>` | `BLOB` |
16//! | `Option<T>` | Same as `T`, but nullable |
17
18use syn::{Field, Type};
19
20pub(crate) struct FieldMeta {
21    /// Whether this field is the primary key of the table. Only one field can
22    /// be marked as the primary key.
23    pub primary: bool,
24
25    /// The name of the field in the database, it equals to the struct field
26    /// name.
27    pub name: String,
28
29    /// The type of the field in Rust.
30    pub ty: syn::Type,
31
32    /// The SQL type of the field, which is derived from the Rust type. This is
33    /// used in the table generator syntax.
34    pub sql_type: String,
35
36    /// The SQL type of the field, without `NOT NULL`.
37    pub raw_sql_type: String,
38
39    /// If the field originates from an `Option<T>`.
40    pub optional: bool,
41}
42
43impl FieldMeta {
44    /// Converts the field metadata into a SQL field definition, which is used
45    /// in the table generator syntax.
46    pub fn into_field_definition(&self) -> String {
47        let mut definition = format!("{} {}", self.name, self.sql_type);
48        if self.primary {
49            definition.push_str(" PRIMARY KEY");
50        }
51        definition
52    }
53}
54
55/// Converts a Rust type into an SQL type string. This is used for generating the
56/// Returns: Type string and whether it's nullable (Option<T>).
57fn rust_type_to_sql_type(ty: &syn::Type) -> (String, bool) {
58    let mut nullable = false;
59    let type_string = match ty {
60        Type::Path(type_path) => {
61            let segment = type_path.path.segments.last().unwrap();
62            match segment.ident.to_string().as_str() {
63                "i8" | "i16" | "i32" | "i64" | "u8" | "u16" | "u32" | "u64" | "isize"
64                | "usize" => "INTEGER".to_string(),
65                "f32" | "f64" => "REAL".to_string(),
66                "String" => "TEXT".to_string(),
67                "bool" => "BOOLEAN".to_string(),
68                "Uuid" => "TEXT".to_string(),
69                "Option" if !segment.arguments.is_empty() => {
70                    nullable = true;
71
72                    let generic_ty = match &segment.arguments {
73                        syn::PathArguments::AngleBracketed(args) if args.args.len() == 1 => {
74                            match &args.args[0] {
75                                syn::GenericArgument::Type(inner_ty) => inner_ty,
76                                _ => panic!(
77                                    "Unsupported generic argument in Option: {:?}",
78                                    args.args[0]
79                                ),
80                            }
81                        }
82                        _ => panic!("Unsupported arguments in Option: {:?}", segment.arguments),
83                    };
84
85                    rust_type_to_sql_type(generic_ty).0
86                }
87                "Vec" if segment.arguments.is_empty() => "BLOB".to_string(),
88                _ => panic!("Unsupported field type: {}", segment.ident),
89            }
90        }
91        _ => panic!("Unsupported field type: {:?}", ty),
92    };
93
94    (type_string, nullable)
95}
96
97impl From<Field> for FieldMeta {
98    fn from(field: Field) -> Self {
99        let primary = field
100            .attrs
101            .iter()
102            .any(|attr| attr.meta.path().is_ident("primary"));
103        let name = field.ident.as_ref().unwrap().to_string();
104        let ty = field.ty.clone();
105        let (sql_type, optional) = rust_type_to_sql_type(&field.ty);
106
107        FieldMeta {
108            primary,
109            name,
110            ty,
111            sql_type: if optional { sql_type.clone() } else { format!("{} NOT NULL", sql_type.clone()) },
112            raw_sql_type: sql_type,
113            optional,
114        }
115    }
116}
