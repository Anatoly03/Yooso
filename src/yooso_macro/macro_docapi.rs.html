29/// The [docapi][super::docapi] attribute marks the function as an endpoint and generates
30/// documentation for the API.
31///
32/// # Example
33///
34/// ```no_run,no_test,ignore
35/// #[docapi()]
36/// #[get("/view/<uuid>")]
37/// pub async fn view_component(..., uuid: &str) -> ... {
38///     ...
39/// }
40/// ```
41pub fn docapi_fn(_meta: DocApiMeta, func: ItemFn) -> TokenStream {
42    // The "root" of the rust crate, which is a member of the workspace. The openapi
43    // documentation will be generated into this path + "openapi".
44    let openapi_dir = caller_openapi_dir();
45
46    // The build hash.
47    let build_hash = get_build_hash();
48
49    let url_attr =
50        get_rocket_attr(&func).expect("Expected a Rocket HTTP method attribute on the function.");
51    let http_method = url_attr.path().get_ident().unwrap().to_string();
52
53    // Rocket URL attributes are of the form `#[get("/path")]` or `#[post("/path", body = "<body>")]`.
54    // Te path is the first argument to the attribute, which is a string literal.
55    let url = match &url_attr.meta {
56        Meta::List(meta_list) => meta_list
57            .tokens
58            .clone()
59            .into_iter()
60            .next()
61            .expect("Expected a Rocket URL attribute to have at least one argument.")
62            .to_string()
63            .trim_matches('"')
64            .to_string(),
65        _ => panic!("Expected a Rocket URL attribute to be a list."),
66    };
67
68    // Open the file and create it if not exists. We also lock it in a cross-process
69    // lock to prevent side effects.
70    //
71    // ...
72    // ├── openapi
73    // │   ├── ...
74    // │   └── openapi.json
75    // └ ...
76    let mut lock = lock_file(openapi_dir.join("openapi.json"));
77    let mut guard = lock.write().unwrap();
78
79    // Parse or create the JSON structure
80    let mut openapi_value = read_json_file(&mut guard).unwrap_or_else(|_| {
81        json!({
82            "build_hash": build_hash,
83            "paths": {},
84            "components": {},
85        })
86    });
87
88    // Check for build hash match (to ensure file is cleaned once before a new build)
89    match openapi_value.get("build_hash").and_then(|v| v.as_str()) {
90        Some(existing_hash) if existing_hash != build_hash => {
91            // This is a new build - we keep the endpoints but update the hash
92            openapi_value["build_hash"] = json!(build_hash);
93        }
94        None => {
95            // No hash found - add it
96            openapi_value["build_hash"] = json!(build_hash);
97        }
98        _ => { /* hash matches */ }
99    }
100
101    // Set OpenAPI file metadata.
102    openapi_value["openapi"] = json!("3.1.0");
103    openapi_value["info"] = json!({
104        "title": std::env::var("CARGO_PKG_NAME").unwrap(),
105        "version": std::env::var("CARGO_PKG_VERSION").unwrap(),
106        // "authors": std::env::var("CARGO_PKG_AUTHORS").unwrap().split(':').collect::<Vec<_>>(),
107    });
108
109    // Ensure paths and components exist. This is a safety check in case the file
110    // was manually edited and these keys were removed.
111    if !openapi_value.get("paths").is_some() {
112        openapi_value["paths"] = json!({});
113    }
114    if !openapi_value.get("components").is_some() {
115        openapi_value["components"] = json!({});
116    }
117
118    // Add the endpoint to the paths
119    // Note: Use the path as key (without query parameters)
120    let path_key = {
121        let path_prefix = url.split('?').next().unwrap_or(&url);
122        let mut pk = String::new();
123
124        for segment in path_prefix.split('/') {
125            if !segment.is_empty() {
126                pk.push('/');
127            }
128
129            if segment.starts_with('<') && segment.ends_with('>') {
130                pk.push_str(format!("{{{}}}", &segment[1..segment.len() - 1]).as_str());
131            } else {
132                pk.push_str(segment);
133            }
134        }
135
136        pk
137    };
138    // let path_key = url.split('?').next().unwrap_or(&url);
139
140    // The `operationId` field. It's supposed to remove the leading "api" from the
141    // path as well as remove inputs in angle brackets. For example, `GET /api/view/<uuid>/list`
142    // will produce `get_view_list`
143    let operation_id = {
144        let mut oid = http_method.to_lowercase();
145
146        for segment in path_key.split('/') {
147            if !segment.is_empty()
148                && segment != "api"
149                && !segment.starts_with('{')
150                && !segment.ends_with('}')
151            {
152                oid.push('_');
153                oid.push_str(segment);
154            }
155        }
156
157        oid
158    };
159
160    // Ensure the path exists.
161    if !openapi_value["paths"].get(&path_key).is_some() {
162        openapi_value["paths"][&path_key] = json!({});
163    }
164
165    // Process path parameter names.
166    let path_params = path_key
167        .split('/')
168        .filter(|s| !s.is_empty())
169        .filter(|segment| segment.starts_with('{') && segment.ends_with('}'))
170        .map(|segment| &segment[1..segment.len() - 1])
171        .collect::<Vec<_>>();
172
173    // Path parameter documentation.
174    let path_param_docs = path_params
175        .iter()
176        .map(|param| {
177            json!({
178                "name": param,
179                "in": "path",
180                "required": true,
181                "schema": {
182                    "type": "string",
183                },
184                "description": format!("To-do: document `{}` path parameter", param),
185            })
186        })
187        .collect::<Vec<_>>();
188
189    // Add the HTTP method with its operation
190    let method_lower = http_method.to_lowercase();
191    openapi_value["paths"][&path_key][&method_lower] = json!({
192        "operationId": operation_id,
193        "summary": format!("To-do: document `{}` endpoint", &url),
194        "parameters": path_param_docs
195    });
196
197    // IMPORTANT: Write the file - ONLY truncate when creating a new file
198    // For existing files, we should NOT truncate
199    guard
200        .seek(SeekFrom::Start(0))
201        .expect("Failed to seek to start");
202    guard.set_len(0).expect("Failed to truncate");
203    guard
204        .write_all(openapi_value.to_string().as_bytes())
205        .expect("Failed to write");
206
207    // Print the endpoint
208    print_documentation(&http_method, &url);
209    func.to_token_stream()
210}

212/// The [docapi][super::docapi] attribute marks the struct as a component and
213/// generates documentation for the API.
214///
215/// # Example
216///
217/// ```no_run,no_test,ignore
218/// #[docapi()]
219/// pub struct Entity {
220///     /// The UUID of the component to view.
221///     uuid: Uuid,
222///
223///     // The timestamp of the component creation.
224///     created_at: i32,
225/// }
226/// ```
227pub fn docapi_struct(_meta: DocApiMeta, strucc: ItemStruct) -> TokenStream {
228    // The "root" of the rust crate, which is a member of the workspace. The openapi
229    // documentation will be generated into this path + "openapi".
230    let openapi_dir = caller_openapi_dir();
231
232    // The build hash.
233    let build_hash = get_build_hash();
234
235    // Open the file and create it if not exists. We also lock it in a cross-process
236    // lock to prevent side effects.
237    //
238    // ...
239    // ├── openapi
240    // │   ├── ...
241    // │   └── openapi.json
242    // └ ...
243    let mut lock = lock_file(openapi_dir.join("openapi.json"));
244    let mut guard = lock.write().unwrap();
245
246    // Parse or create the JSON structure
247    let mut openapi_value = read_json_file(&mut guard).unwrap_or_else(|_| {
248        json!({
249            "build_hash": build_hash,
250            "paths": {},
251            "components": {},
252        })
253    });
254
255    // Check for build hash match (to ensure file is cleaned once before a new build)
256    match openapi_value.get("build_hash").and_then(|v| v.as_str()) {
257        Some(existing_hash) if existing_hash != build_hash => {
258            // This is a new build - we keep the endpoints but update the hash
259            openapi_value["build_hash"] = json!(build_hash);
260        }
261        None => {
262            // No hash found - add it
263            openapi_value["build_hash"] = json!(build_hash);
264        }
265        _ => { /* hash matches */ }
266    }
267
268    // Set OpenAPI file metadata.
269    openapi_value["openapi"] = json!("3.1.0");
270    openapi_value["info"] = json!({
271        "title": std::env::var("CARGO_PKG_NAME").unwrap(),
272        "version": std::env::var("CARGO_PKG_VERSION").unwrap(),
273        // "authors": std::env::var("CARGO_PKG_AUTHORS").unwrap().split(':').collect::<Vec<_>>(),
274    });
275
276    // Ensure paths and components exist. This is a safety check in case the file
277    // was manually edited and these keys were removed.
278    if !openapi_value.get("paths").is_some() {
279        openapi_value["paths"] = json!({});
280    }
281    if !openapi_value.get("components").is_some() {
282        openapi_value["components"] = json!({});
283    }
284
285    // Add the schema to the components.
286    let struct_name = strucc.ident.to_string();
287
288    // Determine the required fields of the struct. A field is required if it is not an Option<T>.
289    let required_struct_fields = strucc
290        .fields
291        .iter()
292        .filter_map(|s| match &s.ty {
293            Type::Path(TypePath { path, .. })
294                if path.segments.iter().any(|f| f.ident == "Option") =>
295            {
296                None
297            }
298            _ => s.ident.as_ref().map(|f| f.to_string()),
299        })
300        .collect::<Vec<_>>();
301
302    // Get the properties of every struct field.
303    let struct_properties = strucc
304        .fields
305        .iter()
306        .map(|field| {
307            let ty = get_openapi_type(&field.ty);
308            let format = get_openapi_format(&field);
309            let description = field
310                .attrs
311                .iter()
312                .filter(|attr| attr.path().is_ident("doc"))
313                .filter_map(|attr| match &attr.meta {
314                    Meta::NameValue(meta) => meta
315                        .value
316                        .to_token_stream()
317                        .to_string()
318                        .trim_matches('"')
319                        .to_string()
320                        .into(),
321                    _ => None,
322                })
323                .collect::<Vec<_>>();
324            // let docapi_attr = field
325            //     .attrs
326            //     .iter()
327            //     .find(|attr| attr.path().is_ident("docapi"))
328            //     .expect("Expected a #[docapi] attribute on the struct field.");
329            let key = field
330                .ident
331                .as_ref()
332                .expect("Only named identifiers allowed")
333                .to_string();
334            let mut value = json!({
335                "type": ty,
336                "description": description.join(" "),
337            });
338
339            if let Some(format) = format {
340                value["format"] = json!(format);
341            }
342
343            (key, value)
344        })
345        .collect::<HashMap<_, _>>();
346
347    let description = strucc
348        .attrs
349        .iter()
350        .filter(|attr| attr.path().is_ident("doc"))
351        .filter_map(|attr| match &attr.meta {
352            Meta::NameValue(meta) => meta
353                .value
354                .to_token_stream()
355                .to_string()
356                .trim_matches('"')
357                .trim()
358                .to_string()
359                .into(),
360            _ => None,
361        })
362        .collect::<Vec<_>>();
363
364    openapi_value["components"]["schemas"][&struct_name] = json!({
365        "type": "object",
366        "required": required_struct_fields,
367        "properties": struct_properties,
368        "description": description.join(" "),
369    });
370
371    // IMPORTANT: Write the file - ONLY truncate when creating a new file
372    // For existing files, we should NOT truncate
373    guard
374        .seek(SeekFrom::Start(0))
375        .expect("Failed to seek to start");
376    guard.set_len(0).expect("Failed to truncate");
377    guard
378        .write_all(openapi_value.to_string().as_bytes())
379        .expect("Failed to write");
380
381    strucc.to_token_stream()
382}

384/// The openapi documentation will be generated into the root of the project + "openapi".
385/// This function will create the directory if it doesn't exist and return the path to it.
386/// It will also fill the directory with a `.gitignore` file to prevent it from being
387/// staged into version control.
388///
389/// ```txt
390/// api/
391/// ├── openapi/
392/// |   ├── ...
393/// │   └── .gitignore
394/// ├── src/
395/// │  └── ...
396/// └ Cargo.toml
397/// ```
398fn caller_openapi_dir() -> PathBuf {
399    let path = PathBuf::from(std::env::var("CARGO_MANIFEST_DIR").unwrap()).join("openapi");
400
401    // Create the openapi directory if it doesn't exist. This will build the following layout:
402    //
403    // api/
404    // ├── openapi/
405    // │   └── .gitignore
406    // ├── src/
407    // │  └── ...
408    // └ Cargo.toml
409    std::fs::create_dir_all(&path).expect("Failed to create openapi directory.");
410    std::fs::write(&path.join(".gitignore"), "*").expect("Failed to write .gitignore file.");
411
412    path
413}

415/// Concatenation of process ID and current timestamp which is unique per build and
416/// equal for all macro invocations in the same build.
417fn get_build_hash() -> String {
418    // Read the current timestamp of the build. If not set, set it.
419    let timestamp = match std::env::var("YOOSO_BUILD_TIMESTAMP") {
420        Ok(ts) => ts,
421        Err(_) => {
422            let ts = format!("{}", chrono::Utc::now().timestamp_millis());
423
424            unsafe {
425                std::env::set_var("YOOSO_BUILD_TIMESTAMP", &ts);
426            }
427
428            ts
429        }
430    };
431
432    // Concatenate process ID and timestamp.
433    format!("{}-{}", std::process::id(), timestamp)
434}

436/// Open the file and create it if not exists. It will be locked in a cross-process
437/// lock to prevent side effects.
438fn lock_file(path: PathBuf) -> RwLock<File> {
439    let openapi_file = File::options()
440        .read(true)
441        .write(true)
442        .create(true)
443        .open(path)
444        .expect("Failed to open openapi.json file.");
445
446    RwLock::new(openapi_file)
447}

449/// Reads the contents of the file from the given lock guard and parses it as JSON. If the
450/// file is empty or invalid JSON, it will return an error.
451fn read_json_file<V: DeserializeOwned>(guard: &mut RwLockWriteGuard<File>) -> std::io::Result<V> {
452    // Buffer the contents of te file into a string.
453    let mut content = String::new();
454    guard.read_to_string(&mut content)?;
455
456    // Parse as JSON; if it is invalid JSON or an empty file - start fresh
457    serde_json::from_str::<V>(&content).map_err(|_| std::io::ErrorKind::InvalidData.into())
458}

460/// Retrieves the HTTP method from the rocket api function.
461fn get_rocket_attr<'a>(func: &'a ItemFn) -> Option<&'a Attribute> {
462    const HTTP_METHODS: &[&str] = &["get", "post", "put", "delete", "patch"];
463
464    // &func.attrs.iter().filter(|attr| attr.path().get_ident().is_some()).find(|attr| HTTP_METHODS.contains(&attr.path().get_ident().unwrap().to_string().as_str()));
465    for attr in &func.attrs {
466        let ident = match attr.path().get_ident() {
467            Some(ident) => ident,
468            None => continue,
469        };
470
471        if HTTP_METHODS.contains(&ident.to_string().as_str()) {
472            return Some(attr);
473        }
474    }
475
476    None
477}

479/// Returns the OpenAPI type for a given Rust type. This is used to generate the
480/// `type` field in the OpenAPI schema for struct fields.
481fn get_openapi_type(ty: &Type) -> &str {
482    match &ty {
483        Type::Array(_) => "array",
484        // Type::BareFn(type_bare_fn) => todo!(),
485        Type::Group(type_group) => get_openapi_type(&type_group.elem),
486        // Type::ImplTrait(type_impl_trait) => todo!(),
487        // Type::Infer(type_infer) => todo!(),
488        // Type::Macro(type_macro) => todo!(),
489        // Type::Never(type_never) => todo!(),
490        Type::Paren(type_paren) => get_openapi_type(&type_paren.elem),
491        Type::Path(TypePath { path, .. }) => {
492            let last_segment = path
493                .segments
494                .last()
495                .expect("Expected a type path to have at least one segment.");
496            let ident_str = last_segment.ident.to_string();
497            let arguments = &last_segment.arguments;
498
499            match ident_str.as_str() {
500                "String" | "str" => "string",
501                "i8" | "i16" | "i32" | "i64" | "i128" | "isize" => "integer",
502                "u8" | "u16" | "u32" | "u64" | "u128" | "usize" => "integer",
503                "f32" | "f64" => "number",
504                "bool" => "boolean",
505                "Vec" => "array",
506                "Option" => {
507                    let generics = match arguments {
508                        PathArguments::AngleBracketed(angle_bracketed) => &angle_bracketed.args,
509                        _ => return "object",
510                    };
511
512                    if let Some(GenericArgument::Type(inner_ty)) = generics.first() {
513                        return get_openapi_type(inner_ty);
514                    }
515
516                    "object"
517                }
518                "Uuid" => "string",
519                _ => "object", // Default to object for unknown types
520            }
521        }
522        Type::Ptr(type_ptr) => get_openapi_type(&type_ptr.elem),
523        Type::Reference(type_reference) => get_openapi_type(&type_reference.elem),
524        Type::Slice(_) => "array",
525        // Type::TraitObject(type_trait_object) => todo!(),
526        Type::Tuple(type_tuple) => {
527            if type_tuple.elems.is_empty() {
528                "null" // Empty tuple represents the unit type `()`, which is often treated as null
529            } else {
530                "array" // Non-empty tuples can be represented as arrays in OpenAPI
531            }
532        }
533        // Type::Verbatim(token_stream) => todo!(),
534        _ => panic!("Unsupported type for OpenAPI generation: {:?}", ty),
535    }
536}

538/// Returns the OpenAPI format for a given Rust type. This is used to generate the
539/// `format` field in the OpenAPI schema for struct fields. Currently the only functionality
540/// this method provides is to return the correct format for integer types.
541///
542/// ```no_run,no_test,ignore
543/// #[docapi()]
544/// pub struct Auth {
545///     #[docapi(format = "email")]
546///     pub email: String,
547/// }
548/// ```
549///
550/// See: https://docs.bump.sh/openapi/v3.2/data-models/schema-and-data-types/#data-formats
551fn get_openapi_format(field: &Field) -> Option<&str> {
552    match &field.ty {
553        Type::Path(TypePath { path, .. }) => {
554            let last_segment = path
555                .segments
556                .last()
557                .expect("Expected a type path to have at least one segment.");
558            let ident_str = last_segment.ident.to_string();
559
560            match ident_str.as_str() {
561                // isize
562                #[cfg(target_pointer_width = "32")]
563                "isize" => Some("int32"),
564                #[cfg(target_pointer_width = "64")]
565                "isize" => Some("int64"),
566
567                // usize
568                #[cfg(target_pointer_width = "32")]
569                "usize" => Some("int32"),
570                #[cfg(target_pointer_width = "64")]
571                "usize" => Some("int64"),
572
573                // signed integers
574                "i8" | "i16" | "i32" => Some("int32"),
575                "i64" | "i128" => Some("int64"),
576
577                // unsigned integers
578                "u8" | "u16" | "u32" => Some("int32"),
579                "u64" | "u128" => Some("int64"),
580
581                // uuid
582                "Uuid" => Some("uuid"),
583
584                _ => None,
585            }
586        }
587        _ => None,
588    }
589}

591/// Print the documentation for the given API endpoint function to the build logs.
592fn print_documentation(http_method: &str, url: &str) {
593    let mut stdout = StandardStream::stdout(ColorChoice::Always);
594    let endpoint = format!("{} {}", http_method.to_uppercase(), url);
595    print_title_line(&mut stdout, "Endpoint", Color::Cyan, &endpoint).unwrap();
596}

598/// Prints a title line with the given title, color, and body to the provided stdout stream.
599/// This matches the Rust "terminal output" style for printing colored text to the terminal.
600///
601/// ```no_run,no_test,ignore
602/// // Prints:
603/// //
604/// // ```
605/// //              ...
606/// //    Compiling yooso-macro v0.0.0 (/home/anatoly/Projects/Yooso/yooso-macro)
607/// //    Compiling yooso-storage v0.0.0 (/home/anatoly/Projects/Yooso/yooso-storage)
608/// //    Compiling yooso-api v0.0.0 (/home/anatoly/Projects/Yooso/yooso-api)
609/// //     Endpoint /api/todo
610/// //              ...
611/// // ```
612/// print_title_line(&mut stdout, "Endpoint", Color::Cyan, "/api/todo").unwrap();
613/// ```
614fn print_title_line(
615    stdout: &mut StandardStream,
616    title: &str,
617    color: Color,
618    body: &str,
619) -> Result<(), std::io::Error> {
620    colored!(
621        stdout,
622        "{}{}{}{}{}{}\n", // format string
623        fg!(Some(color)),
624        bold!(true),               // title style
625        format!("{:>12} ", title), // title
626        fg!(None),
627        bold!(false), // reset style
628        body
629    )
630}
