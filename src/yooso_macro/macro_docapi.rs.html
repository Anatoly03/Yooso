14/// The [docapi][super::docapi] attribute marks the function as an endpoint and generates
15/// documentation for the API.
16///
17/// # Example
18///
19/// ```no_run
20/// #[docapi()]
21/// #[get("/view/<uuid>")]
22/// pub async fn view_component(..., uuid: &str) -> ... {
23///     ...
24/// }
25/// ```
26pub fn docapi_fn(func: ItemFn) -> TokenStream {
27    let url_attr =
28        get_rocket_attr(&func).expect("Expected a Rocket HTTP method attribute on the function.");
29    let http_method = url_attr.path().get_ident().unwrap().to_string();
30
31    // The "root" of the rust crate, which is a member of the workspace. The openapi
32    // documentation will be generated into this path + "openapi".
33    let caller_openapi_dir = {
34        let s = format!(
35            "{}{}",
36            std::env::var("CARGO_MANIFEST_DIR").unwrap(),
37            "/openapi"
38        );
39        PathBuf::from(s)
40    };
41
42    // Read the current timestamp of the build, which generates the unique build hash. If not
43    // set, set it.
44    let timestamp = match std::env::var("YOOSO_BUILD_TIMESTAMP") {
45        Ok(ts) => ts,
46        Err(_) => {
47            let ts = format!("{}", chrono::Utc::now().timestamp_millis());
48
49            unsafe {
50                std::env::set_var("YOOSO_BUILD_TIMESTAMP", &ts);
51            }
52
53            ts
54        }
55    };
56
57    // The build hash.
58    let build_hash = format!("{}-{}", std::process::id(), timestamp);
59
60    // Shortcut for generating a file path for the openapi documentation.
61    let fp = |filename: &str| caller_openapi_dir.join(filename);
62
63    // Rocket URL attributes are of the form `#[get("/path")]` or `#[post("/path", body = "<body>")]`.
64    // Te path is the first argument to the attribute, which is a string literal.
65    let url = match &url_attr.meta {
66        syn::Meta::List(meta_list) => meta_list
67            .tokens
68            .clone()
69            .into_iter()
70            .next()
71            .expect("Expected a Rocket URL attribute to have at least one argument.")
72            .to_string()
73            .trim_matches('"')
74            .to_string(),
75        _ => panic!("Expected a Rocket URL attribute to be a list."),
76    };
77
78    // Create the openapi directory if it doesn't exist. This will build the following layout:
79    //
80    // api/
81    // ├── openapi/
82    // │   └── .gitignore
83    // ├── src/
84    // │  └── ...
85    // └ Cargo.toml
86    std::fs::create_dir_all(&caller_openapi_dir).expect("Failed to create openapi directory.");
87    std::fs::write(fp(".gitignore"), "*").expect("Failed to write .gitignore file.");
88
89    // Open the file and create it if not exists. We also lock it in a cross-process
90    // lock to prevent side effects.
91    //
92    // api/
93    // ├── openapi/
94    // │   ├── ...
95    // │   └── openapi.json
96    // └ ...
97    let openapi_file = File::options()
98        .read(true)
99        .write(true)
100        .create(true)
101        .open(fp("openapi.json"))
102        .expect("Failed to open openapi.json file.");
103    let mut lock = RwLock::new(openapi_file);
104    let mut guard = lock.write().unwrap();
105
106    // Parse or create the JSON structure
107    let mut openapi_value = {
108        // Buffer the contents of te file into a string.
109        let mut content = String::new();
110        guard
111            .read_to_string(&mut content)
112            .expect("Failed to read openapi.json file.");
113
114        // Parse as JSON; if it is invalid JSON or an empty file - start fresh
115        match serde_json::from_str::<Value>(&content) {
116            Ok(value) => value,
117            Err(_) => {
118                serde_json::json!({
119                    "build_hash": build_hash,
120                    "paths": {},
121                    "components": {},
122                })
123            }
124        }
125    };
126
127    // Check for build hash match (to ensure file is cleaned once before a new build)
128    match openapi_value.get("build_hash").and_then(|v| v.as_str()) {
129        Some(existing_hash) if existing_hash != build_hash => {
130            // This is a new build - we keep the endpoints but update the hash
131            openapi_value["build_hash"] = serde_json::json!(build_hash);
132        }
133        None => {
134            // No hash found - add it
135            openapi_value["build_hash"] = serde_json::json!(build_hash);
136        }
137        _ => { /* hash matches */ }
138    }
139
140    // Set OpenAPI file metadata.
141    openapi_value["openapi"] = serde_json::json!("3.1.0");
142    openapi_value["info"] = serde_json::json!({
143        "title": std::env::var("CARGO_PKG_NAME").unwrap(),
144        "version": std::env::var("CARGO_PKG_VERSION").unwrap(),
145        "authors": std::env::var("CARGO_PKG_AUTHORS").unwrap().split(':').collect::<Vec<_>>(),
146    });
147
148    // Ensure paths and components exist. This is a safety check in case the file
149    // was manually edited and these keys were removed.
150    if !openapi_value.get("paths").is_some() {
151        openapi_value["paths"] = serde_json::json!({});
152    }
153    if !openapi_value.get("components").is_some() {
154        openapi_value["components"] = serde_json::json!({});
155    }
156
157    // Add the endpoint to the paths
158    // Note: Use the path as key (without query parameters)
159    let path_key = {
160        let path_prefix = url.split('?').next().unwrap_or(&url);
161        let mut pk = String::new();
162
163        for segment in path_prefix.split('/') {
164            if !segment.is_empty() {
165                pk.push('/');
166            }
167
168            if segment.starts_with('<') && segment.ends_with('>') {
169                pk.push_str(format!("{{{}}}", &segment[1..segment.len() - 1]).as_str());
170            } else {
171                pk.push_str(segment);
172            }
173        }
174
175        pk
176    };
177    // let path_key = url.split('?').next().unwrap_or(&url);
178
179    // The `operationId` field. It's supposed to remove the leading "api" from the
180    // path as well as remove inputs in angle brackets. For example, `GET /api/view/<uuid>/list`
181    // will produce `get_view_list`
182    let operation_id = {
183        let mut oid = http_method.to_lowercase();
184
185        for segment in path_key.split('/') {
186            if !segment.is_empty()
187                && segment != "api"
188                && !segment.starts_with('{')
189                && !segment.ends_with('}')
190            {
191                oid.push('_');
192                oid.push_str(segment);
193            }
194        }
195
196        oid
197    };
198
199    // Ensure the path exists.
200    if !openapi_value["paths"].get(&path_key).is_some() {
201        openapi_value["paths"][&path_key] = serde_json::json!({});
202    }
203
204    // Add the HTTP method with its operation
205    let method_lower = http_method.to_lowercase();
206    openapi_value["paths"][&path_key][&method_lower] = serde_json::json!({
207        "operationId": operation_id,
208        "summary": "Todo: Implement #[docapi(summary = \"...\")]",
209    });
210
211    // IMPORTANT: Write the file - ONLY truncate when creating a new file
212    // For existing files, we should NOT truncate
213    guard
214        .seek(SeekFrom::Start(0))
215        .expect("Failed to seek to start");
216    guard.set_len(0).expect("Failed to truncate");
217    guard
218        .write_all(openapi_value.to_string().as_bytes())
219        .expect("Failed to write");
220
221    // Print the endpoint
222    print_documentation(&http_method, &url);
223    func.to_token_stream()
224}

245/// Retrieves the HTTP method from the rocket api function.
246pub fn get_rocket_attr<'a>(func: &'a ItemFn) -> Option<&'a Attribute> {
247    const HTTP_METHODS: &[&str] = &["get", "post", "put", "delete", "patch"];
248
249    // &func.attrs.iter().filter(|attr| attr.path().get_ident().is_some()).find(|attr| HTTP_METHODS.contains(&attr.path().get_ident().unwrap().to_string().as_str()));
250    for attr in &func.attrs {
251        let ident = match attr.path().get_ident() {
252            Some(ident) => ident,
253            None => continue,
254        };
255
256        if HTTP_METHODS.contains(&ident.to_string().as_str()) {
257            return Some(attr);
258        }
259    }
260
261    None
262}

264/// Print the documentation for the given API endpoint function to the build logs.
265fn print_documentation(http_method: &str, url: &str) {
266    let mut stdout = StandardStream::stdout(ColorChoice::Always);
267    let endpoint = format!("{} {}", http_method.to_uppercase(), url);
268    print_title_line(&mut stdout, "Endpoint", Color::Cyan, &endpoint).unwrap();
269}

271/// Prints a title line with the given title, color, and body to the provided stdout stream.
272/// This matches the Rust "terminal output" style for printing colored text to the terminal.
273///
274/// ```no_run
275/// // Prints:
276/// //
277/// // ```
278/// //              ...
279/// //    Compiling yooso-macro v0.0.0 (/home/anatoly/Projects/Yooso/yooso-macro)
280/// //    Compiling yooso-storage v0.0.0 (/home/anatoly/Projects/Yooso/yooso-storage)
281/// //    Compiling yooso-api v0.0.0 (/home/anatoly/Projects/Yooso/yooso-api)
282/// //     Endpoint /api/todo
283/// //              ...
284/// // ```
285/// print_title_line(&mut stdout, "Endpoint", Color::Cyan, "/api/todo").unwrap();
286/// ```
287fn print_title_line(
288    stdout: &mut StandardStream,
289    title: &str,
290    color: Color,
291    body: &str,
292) -> Result<(), std::io::Error> {
293    colored!(
294        stdout,
295        "{}{}{}{}{}{}\n", // format string
296        fg!(Some(color)),
297        bold!(true),               // title style
298        format!("{:>12} ", title), // title
299        fg!(None),
300        bold!(false), // reset style
301        body
302    )
