5/// General error type for Yooso, which is used in database management as
6/// well as API-level handlers.
7#[derive(Debug)]
8#[non_exhaustive]
9pub enum Error {
10    /// An error originating from the Rusqlite database layer.
11    RusqliteError(::rusqlite::Error),
12
13    /// An error originating from a poisoned mutex. We store a short
14    /// diagnostic string (the `PoisonError` does not expose a useful
15    /// typed payload here across guard types), and provide conversions
16    /// from `PoisonError<T>`.
17    MutexPoisoned(String),
18
19    /// An error originating from the Uuid library. This typically occurs
20    /// during Uuid parsing.
21    UuidError(::uuid::Error),
22
23    /// An error originating from invalid input data, such as as invalid
24    /// component name or field metadata.
25    ValidationError(::util_validation::ValidationError),
26
27    /// A constant for the `404 Not Found`` error code, which is commonly used across
28    /// API handlers.
29    NotFound,
30
31    // /// A generic error type unwrapping into a [rocket] [Status] code.
32    // Code(Status),
33}
34
35/// A typedef of the result returned by Yooso functions.
36pub type Result<T, E = Error> = std::result::Result<T, E>;
