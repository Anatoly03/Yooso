1//! Utility methods for validating field data from input.
2
3use regex::Regex;
4use yooso_core::Error::ValidationError;
5use yooso_core::error::Result;
6
7/// Validates that the trimmed string value is not empty.
8pub fn not_empty(value: &str, field_name: String) -> Result<()> {
9    if value.trim().is_empty() {
10        return Err(ValidationError(format!("{} cannot be empty", field_name)));
11    }
12
13    Ok(())
14}
15
16/// Validates that the string value is a valid SQL identifier. This is a
17/// simple regex check that allows only letters, numbers and underscores, and
18/// does not allow starting with a number.
19pub fn valid_sql_ident(value: &str, field_name: String) -> Result<()> {
20    let re = Regex::new(r"^[a-zA-Z_][a-zA-Z0-9_]*$").unwrap();
21
22    if !re.is_match(value) {
23        return Err(ValidationError(format!(
24            "{} must be a valid SQL identifier (only letters, numbers and underscores, and cannot start with a number)",
25            field_name
26        )));
27    }
28
29    Ok(())
30}
31
32/// Validates that the string value is not a SQL keyword. This is a simple check
33/// against a hardcoded list of common SQL keywords. This is not exhaustive, but
34/// should cover the most common cases.
35pub fn not_sql_keyword(value: &str, field_name: String) -> Result<()> {
36    const SQL_KEYWORDS: &[&str] = &[
37        "select", "from", "where", "insert", "update", "delete", "create", "drop",
38        "alter", "table", "index", "view", "trigger", "procedure", "function",
39        "join", "on", "as", "and", "or", "not", "null", "in", "is", "like",
40        "group", "by", "order", "having", "distinct",
41    ];
42
43    if SQL_KEYWORDS.contains(&value.to_lowercase().as_str()) {
44        return Err(ValidationError(format!(
45            "{} cannot be a SQL keyword `{}`",
46            field_name, value
47        )));
48    }
49
50    Ok(())
51}
